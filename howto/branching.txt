# aby stworzyc branch i kontynuowac prace w nowym branchu nalezy wykonac jedna komende:

git checkout -b <mybranch>





# garsc komend dotyczacych branchowania (polecam rowniez przeczytanie linkow znajdujacych sie w good_to_know.txt):

# Create a new branch called <mybranch>:

git branch <mybranch>

# Make <mybranch> the active branch:

git checkout <mybranch>

# switch back to "master" branch (changes in <mybranch> are not lost):

git checkout master

# upload branch to github:

git push origin <mybranch>

# merging branch <mybranch> to master (pewnie najpierw trzeba zrobic git checkout master):

git merge <mybranch>

# removing unnecessary branch:

git branch -d <mybranch>
